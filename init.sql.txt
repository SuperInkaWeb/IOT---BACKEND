DROP TABLE IF EXISTS alerta CASCADE;
DROP TABLE IF EXISTS configuracion_metrica CASCADE;
DROP TABLE IF EXISTS lectura_sensor CASCADE;
DROP TABLE IF EXISTS sensor CASCADE;
DROP TABLE IF EXISTS suscripcion CASCADE;
DROP TABLE IF EXISTS usuario CASCADE;

DROP TABLE IF EXISTS plan CASCADE;

-- =============================================
-- 2. TABLAS MAESTRAS (Sin dependencias)
-- =============================================

-- 1. Tablas Base
CREATE TABLE plan (
    id SERIAL PRIMARY KEY,
    nombre varchar(100),
    descripcion text,
    precio_mensual numeric(10,2), 
    limite_sensores integer,
    tiene_ia boolean DEFAULT FALSE,
    tiene_reportes_pdf boolean DEFAULT FALSE,
    dias_historial integer
);

CREATE TABLE usuario (
    id SERIAL PRIMARY KEY,
    empresa_id integer,
    plan_id integer,
    nombre varchar(150) NOT NULL,
    email varchar(150) UNIQUE NOT NULL,
    password_hash TEXT,
    rol varchar(20) DEFAULT 'VISOR',
    tipo_usuario varchar(20),
    recibir_alertas_email boolean DEFAULT TRUE,
    activo boolean DEFAULT TRUE,
    fecha_creacion timestamp
);

CREATE TABLE empresa (
    id SERIAL PRIMARY KEY,
    nombre varchar(150) NOT NULL,
    ruc varchar(20) UNIQUE,
    creador_id integer,
    plan_id integer NOT NULL,
    activa boolean DEFAULT TRUE,
    fecha_creacion timestamp
);

CREATE TABLE sensor (
    id SERIAL PRIMARY KEY,
    empresa_id integer,
    usuario_id integer,
    device_id varchar(100) UNIQUE NOT NULL,
    tipo varchar(100) NOT NULL,
    modelo varchar(100),
    ubicacion varchar(150),
    latitud numeric(9,6),
    longitud numeric(9,6),
    activo boolean DEFAULT TRUE,
    fecha_instalacion date NOT NULL,
    es_global boolean DEFAULT FALSE,
    altura_instalacion numeric(3,2)
);

CREATE TABLE suscripcion (
    id SERIAL PRIMARY KEY,
    usuario_id integer NOT NULL,
    empresa_id integer,
    plan_id integer NOT NULL,
    fecha_inicio date,
    fecha_fin date,
    payme_order_id varchar(100),
    estado_pago varchar(20),
    moneda varchar(20) DEFAULT 'PEN',
    estado varchar(20) DEFAULT 'ACTIVA'
);

CREATE TABLE lectura_sensor (
    id BIGSERIAL PRIMARY KEY,
    sensor_id integer NOT NULL,
    tipo_metrica varchar(50) NOT NULL,
    valor double precision NOT NULL,
    temperatura double precision,
    humedad double precision,
    unidad varchar(20),
    fecha timestamp,
    anomalia_detectada boolean DEFAULT FALSE
);

CREATE TABLE configuracion_metrica (
    id SERIAL PRIMARY KEY,
    sensor_id integer NOT NULL,
    tipo_metrica varchar(50) NOT NULL,
    rango_min double precision,
    rango_max double precision,
    unidad varchar(20)
);

CREATE TABLE alerta (
    id BIGSERIAL PRIMARY KEY,
    sensor_id integer NOT NULL,
    lectura_id bigint,
    tipo_metrica varchar(50),
    valor double precision,
    mensaje TEXT,
    nivel varchar(20), -- BAJO, MEDIO, ALTO
    atendida boolean DEFAULT FALSE,
    fecha timestamp
);


-- RELACIONES (FOREIGN KEYS)

ALTER TABLE usuario ADD CONSTRAINT fk_u_empresa FOREIGN KEY (empresa_id) REFERENCES empresa(id);
ALTER TABLE usuario ADD CONSTRAINT fk_u_plan FOREIGN KEY (plan_id) REFERENCES plan(id);
ALTER TABLE empresa ADD CONSTRAINT fk_e_creador FOREIGN KEY (creador_id) REFERENCES usuario(id);
ALTER TABLE empresa ADD CONSTRAINT fk_e_plan FOREIGN KEY (plan_id) REFERENCES plan(id);
ALTER TABLE sensor ADD CONSTRAINT fk_s_empresa FOREIGN KEY (empresa_id) REFERENCES empresa(id);
ALTER TABLE sensor ADD CONSTRAINT fk_s_usuario FOREIGN KEY (usuario_id) REFERENCES usuario(id);
ALTER TABLE suscripcion ADD CONSTRAINT fk_sub_usuario FOREIGN KEY (usuario_id) REFERENCES usuario(id);
ALTER TABLE suscripcion ADD CONSTRAINT fk_sub_empresa FOREIGN KEY (empresa_id) REFERENCES empresa(id);
ALTER TABLE suscripcion ADD CONSTRAINT fk_sub_plan FOREIGN KEY (plan_id) REFERENCES plan(id);
