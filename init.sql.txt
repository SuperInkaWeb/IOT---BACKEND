-- ========= EMPRESA =========
CREATE TABLE empresa (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(150) NOT NULL,
    ruc VARCHAR(20) UNIQUE,
    email_contacto VARCHAR(150),
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ========= PLAN =========
CREATE TABLE plan (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    precio_mensual NUMERIC(10,2),
    limite_sensores INTEGER,
    descripcion TEXT
);

-- ========= SUSCRIPCION =========
CREATE TABLE suscripcion (
    id SERIAL PRIMARY KEY,
    empresa_id INTEGER NOT NULL,
    plan_id INTEGER NOT NULL,
    fecha_inicio DATE DEFAULT CURRENT_DATE,
    fecha_fin DATE,
    estado VARCHAR(20) DEFAULT 'ACTIVA',

    CONSTRAINT fk_suscripcion_empresa
        FOREIGN KEY (empresa_id) REFERENCES empresa(id)
        ON DELETE CASCADE,

    CONSTRAINT fk_suscripcion_plan
        FOREIGN KEY (plan_id) REFERENCES plan(id)
);

-- ========= USUARIO =========
CREATE TABLE usuario (
    id SERIAL PRIMARY KEY,
    empresa_id INTEGER NOT NULL,
    nombre VARCHAR(150) NOT NULL,
    email VARCHAR(150) NOT NULL UNIQUE,
    password_hash TEXT NOT NULL,
    rol VARCHAR(20) DEFAULT 'USER',
    activo BOOLEAN DEFAULT TRUE,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_usuario_empresa
        FOREIGN KEY (empresa_id) REFERENCES empresa(id)
        ON DELETE CASCADE
);

-- ========= SENSOR =========
CREATE TABLE sensor (
    id SERIAL PRIMARY KEY,
    empresa_id INTEGER NOT NULL,
    tipo VARCHAR(100) NOT NULL, -- AIRE, AGUA, ENERGIA
    modelo VARCHAR(100),
    ubicacion VARCHAR(150),
    latitud DECIMAL(9,6),
    longitud DECIMAL(9,6),
    activo BOOLEAN DEFAULT TRUE,
    fecha_instalacion DATE DEFAULT CURRENT_DATE,

    CONSTRAINT fk_sensor_empresa
        FOREIGN KEY (empresa_id) REFERENCES empresa(id)
        ON DELETE CASCADE
);

-- ========= LECTURA SENSOR =========
CREATE TABLE lectura_sensor (
    id BIGSERIAL PRIMARY KEY,
    sensor_id INTEGER NOT NULL,
    tipo_metrica VARCHAR(50) NOT NULL, -- PM2.5, CO2, TEMP, HUMEDAD, PH, etc.
    valor DOUBLE PRECISION NOT NULL,
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_lectura_sensor
        FOREIGN KEY (sensor_id) REFERENCES sensor(id)
        ON DELETE CASCADE
);

-- ========= CONFIGURACION METRICA =========
CREATE TABLE configuracion_metrica (
    id SERIAL PRIMARY KEY,
    sensor_id INTEGER NOT NULL,
    tipo_metrica VARCHAR(50) NOT NULL,
    rango_min DOUBLE PRECISION,
    rango_max DOUBLE PRECISION,
    unidad VARCHAR(20),

    CONSTRAINT fk_config_metrica_sensor
        FOREIGN KEY (sensor_id) REFERENCES sensor(id)
        ON DELETE CASCADE
);

-- ========= ALERTA =========
CREATE TABLE alerta (
    id BIGSERIAL PRIMARY KEY,
    sensor_id INTEGER NOT NULL,
    lectura_id BIGINT,
    tipo_metrica VARCHAR(50),
    valor DOUBLE PRECISION,
    mensaje TEXT,
    nivel VARCHAR(20), -- BAJO, MEDIO, ALTO
    atendida BOOLEAN DEFAULT FALSE,
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_alerta_sensor
        FOREIGN KEY (sensor_id) REFERENCES sensor(id)
        ON DELETE CASCADE,

    CONSTRAINT fk_alerta_lectura
        FOREIGN KEY (lectura_id) REFERENCES lectura_sensor(id)
        ON DELETE SET NULL
);


-- INDICES 


CREATE INDEX idx_sensor_empresa ON sensor(empresa_id);
CREATE INDEX idx_lectura_sensor ON lectura_sensor(sensor_id);
CREATE INDEX idx_lectura_fecha ON lectura_sensor(fecha);
CREATE INDEX idx_alerta_sensor ON alerta(sensor_id);
CREATE INDEX idx_usuario_empresa ON usuario(empresa_id);
